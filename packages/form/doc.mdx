import { Form } from './Form'

## use case

### string

<Form
  schema={{
    type: 'string',
    title: '姓名',
    description: '请输入姓名',
  }}
/>

### number

<Form
  schema={{
    type: 'number',
    title: '年龄',
    description: '请输入年龄',
  }}
/>

### boolean

<Form
  schema={{
    type: 'boolean',
    title: '婚否',
    description: '请输入婚否',
  }}
/>

### array

<Form
  schema={{
    type: 'array',
    items: {
      type: 'string',
      title: '别名',
      description: '请输入别名',
    },
  }}
/>

### object

<Form
  schema={{
    type: 'object',
    properties: {
      name: {
        type: 'string',
        title: '姓名',
        description: '请输入姓名',
      },
      age: {
        type: 'number',
        title: '年龄',
        description: '请输入年龄',
      },
      marital: {
        type: 'boolean',
        title: '婚否',
        description: '请输入婚否',
      },
    },
  }}
/>

### array + object

<Form
  schema={{
    type: 'array',
    title: '标题',
    description: '描述',
    items: {
      type: 'object',
      properties: {
        x: {
          type: 'number',
          title: '横坐标',
          description: '请输入横坐标',
        },
        y: {
          type: 'number',
          title: '纵坐标',
          description: '请输入纵坐标',
        },
      },
    },
  }}
/>
